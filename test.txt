# הגדרת נתיבים לקבצים *המחולקים* שיצרנו עם prepare_fixed_split.py
TRAIN_DATA_DIR = "fixed_split_data/train_data"
VAL_DATA_DIR = "fixed_split_data/validation_data"

# מפות קבצים עבור כל קבוצה
TRAIN_DATA_MAP = {
    LABEL_STEPS:   {'path': [os.path.join(TRAIN_DATA_DIR, 'train_man.csv')]},
    LABEL_VEHICLE: {'path': [os.path.join(TRAIN_DATA_DIR, 'train_car.csv')]},
    LABEL_NOISE:   {'path': [os.path.join(TRAIN_DATA_DIR, 'train_nothing.csv')]}
}

VAL_DATA_MAP = {
    LABEL_STEPS:   {'path': [os.path.join(VAL_DATA_DIR, 'validation_man.csv')]},
    LABEL_VEHICLE: {'path': [os.path.join(VAL_DATA_DIR, 'validation_car2.csv')]},
    LABEL_NOISE:   {'path': [os.path.join(VAL_DATA_DIR, 'validation_nothing2.csv')]}
}

# --- טעינת נתוני אימון ---
print("\n=== Loading Prepared Training Dataset ===")
# הפונקציה load_data_hybrid_by_file צריכה להיות מוגדרת *לפני* קטע זה
train_specs, train_stat_feats, train_labels, _ = load_data_hybrid_by_file(TRAIN_DATA_MAP)
if train_specs.size == 0:
    print("CRITICAL ERROR: Failed to load training dataset. Exiting.")
    exit()

# --- טעינת נתוני וולידציה ---

    print("CRITICAL ERROR: Failed to load validation dataset. Exiting.")
    exit()
