def pad_channels(data, target_channels):
    if data.ndim == 1: data = data.reshape(-1, 1)
    current_channels = data.shape[1]
    if current_channels < target_channels:
        padding = np.zeros((data.shape[0], target_channels - current_channels), dtype=data.dtype)
        return np.hstack((data, padding))
    elif current_channels > target_channels: return data[:, :target_channels]
    else: return data

def create_windows(signal, label, window_size=window_size, stride=stride):
    X_windows, y_labels = [], []
    if signal.shape[0] < window_size: return [], []
    for start in range(0, signal.shape[0] - window_size + 1, stride):
        X_windows.append(signal[start: start + window_size, :])
        y_labels.append(label)
    return X_windows, y_labels

def create_stft_spectrogram(signal_1d, sr=fs, n_fft=N_FFT, hop_length=HOP_LENGTH):
    try:
        signal_float = signal_1d.astype(np.float32)
        f, t, Sxx = spectrogram(signal_float, fs=sr, nperseg=n_fft, noverlap=n_fft - hop_length, nfft=n_fft)
        freq_idx = np.where(f <= highcut)[0]
        if len(freq_idx) == 0: return None
        Sxx = Sxx[freq_idx, :]
        Sxx_db = 10 * np.log10(Sxx + 1e-9)
        Sxx_norm = (Sxx_db - Sxx_db.min()) / (Sxx_db.max() - Sxx_db.min() + 1e-9)
        return Sxx_norm.astype(np.float32)
    except Exception as e:
        print(f"Error in STFT: {e}")
        return None

def extract_statistical_features(signal_1d, sr=fs):
    try:
        features = []
        features.append(np.mean(signal_1d))
        features.append(np.std(signal_1d))
        features.append(np.sqrt(np.mean(signal_1d**2)))
        features.append(librosa.feature.zero_crossing_rate(y=signal_1d).mean())
        stft_result = librosa.stft(signal_1d.astype(np.float32), n_fft=N_FFT, hop_length=HOP_LENGTH)
        spec_mag = np.abs(stft_result)
        features.append(librosa.feature.spectral_centroid(S=spec_mag, sr=sr).mean())
        features.append(librosa.feature.spectral_bandwidth(S=spec_mag, sr=sr).mean())
        features.append(librosa.feature.spectral_rolloff(S=spec_mag, sr=sr, roll_percent=0.85).mean())
        features.append(librosa.feature.spectral_flatness(S=spec_mag).mean())
        mfccs = librosa.feature.mfcc(y=signal_1d.astype(np.float32), sr=sr, n_mfcc=13)
        features.extend(np.mean(mfccs, axis=1))
        return np.array(features, dtype=np.float32)
    except Exception as e:
        print(f"Error extracting statistical features: {e}")
        num_features = 8 + 13 # גודל הוקטור הבסיסי
        return np.zeros(num_features, dtype=np.float32) # החזר וקטור אפסים אם יש בעיה

def load_data_hybrid_by_file(file_map, target_img_size=(IMG_HEIGHT, IMG_WIDTH)):
    all_specs, all_stat_features, all_labels, all_groups = [], [], [], []
    try: b, a = butter(order, [lowcut/(fs/2), highcut/(fs/2)], btype='band')
    except Exception as e: print(f"Filter error: {e}"); return [],[],[],[]

    group_counter = 0
    print("Loading data by file and extracting features...")
    for label, info in file_map.items():
        filepaths = info['path']
        if isinstance(filepaths, str): filepaths = [filepaths]
        for filepath in filepaths:
            try:
                df = pd.read_csv(filepath, header=None)
                df = df.dropna(how='all')
                if df.empty: continue
                imputer = SimpleImputer(strategy='mean')
                data = imputer.fit_transform(df.values) # Use .values
                data = pad_channels(data, max_channels)
                if data.shape[0] < window_size: continue

                windows, labels = create_windows(data, label)
                for w in windows:
                    try:
                        filtered_w = filtfilt(b, a, w, axis=0)
                        signal_to_process = filtered_w[:, 0]
                        spec = create_stft_spectrogram(signal_to_process)
                        if spec is None: continue
                        resized_spec = cv2.resize(spec, (target_img_size[1], target_img_size[0]), interpolation=cv2.INTER_LINEAR)
                        spec_3ch = np.stack([resized_spec]*3, axis=0)

                        # מחשבים תכונות סטט' ומוסיפים MFCC
                        stat_feats_base = extract_statistical_features(signal_to_process)
                        if stat_feats_base.shape[0] != (8+13): # בדיקה אם הפונקציה החזירה את הגודל הצפוי
                             print(f"Warning: Unexpected feature vector size {stat_feats_base.shape[0]} from extract_statistical_features. Skipping window.")
                             continue
                        mfccs_feats = librosa.feature.mfcc(y=signal_to_process.astype(np.float32), sr=fs, n_mfcc=13).mean(axis=1)
                        if mfccs_feats.shape[0] != 13:
                             print(f"Warning: Unexpected MFCC vector size {mfccs_feats.shape[0]}. Skipping window.")
                             continue

                        combined_features = np.concatenate([stat_feats_base, mfccs_feats])
                        if combined_features.shape[0] != NUM_STAT_FEATURES: # בדיקה סופית לגודל
                            print(f"Warning: Final combined feature vector size mismatch {combined_features.shape[0]} vs {NUM_STAT_FEATURES}. Skipping window.")
                            continue

                        all_specs.append(spec_3ch)
                        all_stat_features.append(combined_features)
                        all_labels.append(label)
                        all_groups.append(group_counter)
                    except Exception as inner_e:
                        print(f"Error processing window in file {filepath}: {inner_e}")
                group_counter += 1
            except Exception as e:
                print(f"Error loading file {filepath}: {e}")
    print(f"Total samples loaded: {len(all_specs)}")
    if len(all_specs) > 0:
        return np.array(all_specs, dtype=np.float32), np.array(all_stat_features, dtype=np.float32), np.array(all_labels, dtype=np.int64), np.array(all_groups, dtype=np.int64)
    else:
        return np.array([]), np.array([]), np.array([]), np.array([])

# הגדרת SpecAugment - הפונקציה והקלאס צריכים להיות מוגדרים למעלה
# ... (העתק את הגדרת הקלאס SpecAugment לכאן) ...
class SpecAugment(nn.Module):
    def __init__(self, freq_mask_param=10, time_mask_param=20, num_freq_masks=1, num_time_masks=1):
        super(SpecAugment, self).__init__()
        self.freq_mask_param = freq_mask_param
        self.time_mask_param = time_mask_param
        self.num_freq_masks = num_freq_masks
        self.num_time_masks = num_time_masks

    def forward(self, spec):
        if not isinstance(spec, torch.Tensor): raise TypeError("Input must be a tensor")
        spec_aug = spec.clone(); _, F, T = spec_aug.shape
        for _ in range(self.num_freq_masks):
            f = random.randrange(0, self.freq_mask_param); f0 = random.randrange(0, max(1, F - f))
            spec_aug[:, f0:f0+f, :] = spec_aug.mean()
        for _ in range(self.num_time_masks):
            t = random.randrange(0, self.time_mask_param); t0 = random.randrange(0, max(1, T - t))
            spec_aug[:, :, t0:t0+t] = spec_aug.mean()
        return spec_aug

# הגדרת HybridDataset - הקלאס צריך להיות מוגדר למעלה
# ... (העתק את הגדרת הקלאס HybridDataset לכאן) ...
class HybridDataset(Dataset):
    def __init__(self, specs, stat_feats, labels, transform=None):
        self.specs = specs; self.stat_feats = stat_feats; self.labels = labels; self.transform = transform
    def __len__(self): return len(self.labels)
    def __getitem__(self, idx):
        spec = self.specs[idx]
        # הוספת רעש קטן לתכונות הסטט' גם עוזרת לאוגמנטציה
        stat_feat = self.stat_feats[idx] + np.random.normal(0, 0.01, size=self.stat_feats[idx].shape) # אפשר לשחק עם עוצמת הרעש
        label = self.labels[idx]
        spec_tensor = torch.tensor(spec, dtype=torch.float32)
        if self.transform: spec_tensor = self.transform(spec_tensor)
        stat_tensor = torch.tensor(stat_feat, dtype=torch.float32)
        return spec_tensor, stat_tensor, label

# ... (העתק את הגדרת הקלאס HybridGeoModel לכאן) ...
class HybridGeoModel(nn.Module):
    def __init__(self, dropout_p=0.4, stat_feature_dim=64, combined_dim=128):
        super(HybridGeoModel, self).__init__()
        weights = EfficientNet_B0_Weights.DEFAULT
        self.cnn_backbone = efficientnet_b0(weights=weights)
        original_classifier = self.cnn_backbone.classifier
        if isinstance(original_classifier, nn.Sequential) and len(original_classifier) > 1 and isinstance(original_classifier[1], nn.Linear):
            out_channels = original_classifier[1].in_features
        elif isinstance(original_classifier, nn.Linear):
             out_channels = original_classifier.in_features
        else:
             print("Warning: Could not determine out_channels. Using default 1280.")
             out_channels = 1280 # Fallback for EfficientNet-B0
        print(f"[DEBUG] Determined CNN output features: {out_channels}")
        self.cnn_backbone.classifier = nn.Identity()

        self.stat_mlp = nn.Sequential(
            nn.LayerNorm(NUM_STAT_FEATURES),
            nn.Linear(NUM_STAT_FEATURES, stat_feature_dim * 2), nn.ReLU(), nn.Dropout(dropout_p),
            nn.Linear(stat_feature_dim * 2, stat_feature_dim), nn.ReLU() )

        self.dropout = nn.Dropout(dropout_p)
        self.fc_combined = nn.Linear(out_channels + stat_feature_dim, combined_dim)
        self.fc_output = nn.Linear(combined_dim, NUM_CLASSES)

    def forward(self, spec_input, stat_input):
        cnn_features = self.cnn_backbone(spec_input)
        stat_features = self.stat_mlp(stat_input)
        combined = torch.cat((cnn_features, stat_features), dim=1)
        x = self.dropout(combined); x = F.relu(self.fc_combined(x)); x = self.dropout(x)
        out = self.fc_output(x)
        return out

model = HybridGeoModel(dropout_p=DROPOUT_P, stat_feature_dim=STAT_DIM, combined_dim=COMB_DIM).to(device)

def train_epoch_hybrid(model, dataloader, criterion, optimizer, device, scheduler=None): # הוספת scheduler
    model.train(); running_loss = 0.0; correct = 0; total = 0
    for specs, stat_feats, labels in dataloader:
        specs, stat_feats, labels = specs.to(device), stat_feats.to(device), labels.to(device)
        optimizer.zero_grad()
        outputs = model(specs, stat_feats); loss = criterion(outputs, labels)
        loss.backward()
        # אופציונלי: Gradient Clipping
        # torch.nn.utils.clip_grad_norm_(model.parameters(), max_norm=1.0)
        optimizer.step()
        running_loss += loss.item() * labels.size(0)
        preds = outputs.argmax(dim=1); correct += (preds == labels).sum().item(); total += labels.size(0)
    epoch_loss = running_loss / total; epoch_acc = correct / total
    # אם משתמשים ב-scheduler שמבוסס על epoch
    # if scheduler: scheduler.step()
    return epoch_loss, epoch_acc

def evaluate_model_hybrid(model, dataloader, criterion, device):
    model.eval(); running_loss = 0.0; correct = 0; total = 0
    all_labels = []; all_preds = []
    with torch.no_grad():
        for specs, stat_feats, labels in dataloader:
            specs, stat_feats, labels = specs.to(device), stat_feats.to(device), labels.to(device)
            outputs = model(specs, stat_feats); loss = criterion(outputs, labels)
            running_loss += loss.item() * labels.size(0)
            preds = outputs.argmax(dim=1); correct += (preds == labels).sum().item(); total += labels.size(0)
            all_labels.extend(labels.cpu().numpy()); all_preds.extend(preds.cpu().numpy())
    epoch_loss = running_loss / total; epoch_acc = correct / total
    return epoch_loss, epoch_acc, all_labels, all_preds

def plot_training_results(title, history, true_labels, pred_labels):
    plt.figure(figsize=(12, 5))
    # Loss Plot
    plt.subplot(1, 2, 1)
    plt.plot(history['train_loss'], label='Train Loss')
    plt.plot(history['val_loss'], label='Val Loss')
    plt.title(f"{title} - Loss")
    plt.xlabel("Epoch"); plt.ylabel("Loss"); plt.legend(); plt.grid(True)
    # Accuracy Plot
    plt.subplot(1, 2, 2)
    plt.plot(history['train_acc'], label='Train Acc')
    plt.plot(history['val_acc'], label='Val Acc')
    plt.title(f"{title} - Accuracy")
    plt.xlabel("Epoch"); plt.ylabel("Accuracy"); plt.legend(); plt.grid(True)
    plt.tight_layout()
    plt.savefig(os.path.join(SAVE_PATH_BASE, f"{title}_train_history.png"))
    plt.show()

    # Confusion Matrix
    cm = confusion_matrix(true_labels, pred_labels, labels=range(NUM_CLASSES)) # Ensure all classes appear
    plt.figure(figsize=(6, 5))
    sns.heatmap(cm, annot=True, fmt="d", cmap="Blues",
                xticklabels=CLASS_NAMES, yticklabels=CLASS_NAMES)
    plt.title(f"{title} - Validation Confusion Matrix")
    plt.ylabel("True Label"); plt.xlabel("Predicted Label")
    plt.savefig(os.path.join(SAVE_PATH_BASE, f"{title}_val_cm.png"))
    plt.show()
    print(f"\n--- Validation Classification Report ({title}) ---")
    # הוספת labels=range(NUM_CLASSES) ו-zero_division=0
    print(classification_report(true_labels, pred_labels, target_names=CLASS_NAMES,
                              labels=range(NUM_CLASSES), zero_division=0))
    print("-" * 50)

# =====================================================================================
# =                    לולאת האימון הראשית (ריצה אחת)                                 =
# =====================================================================================
print("\n=== Starting Training ===")

# הגדרת Loss ואופטימייזר
# חישוב משקולות דינמי (אופציונלי, אבל עדיף על סטטי)
# class_counts = np.bincount(y_train_smote)
# class_weights = 1. / torch.tensor(class_counts, dtype=torch.float)
# class_weights = class_weights / class_weights.sum() * NUM_CLASSES # נרמול
# criterion = nn.CrossEntropyLoss(weight=class_weights.to(device))
criterion = nn.CrossEntropyLoss() # נתחיל בלי משקולות, כי SMOTE כבר איזן
optimizer = optim.AdamW(model.parameters(), lr=LEARNING_RATE, weight_decay=WEIGHT_DECAY)

# אופציונלי: Learning Rate Scheduler
scheduler = ReduceLROnPlateau(optimizer, mode='min', factor=0.1, patience=PATIENCE // 2, verbose=True)

best_val_loss = float('inf')
best_model_state = None
epochs_no_improve = 0
history = {'train_loss': [], 'train_acc': [], 'val_loss': [], 'val_acc': []}
all_val_labels, all_val_preds = [], [] # לשמירת התוצאות מה-epoch הטוב ביותר

start_time = time.time()

for epoch in range(1, EPOCHS + 1):
    epoch_start_time = time.time()

    train_loss, train_acc = train_epoch_hybrid(model, train_loader, criterion, optimizer, device) # הסרנו scheduler מכאן
    val_loss, val_acc, val_labels, val_preds = evaluate_model_hybrid(model, val_loader, criterion, device)

    epoch_duration = time.time() - epoch_start_time

    history['train_loss'].append(train_loss)
    history['train_acc'].append(train_acc)
    history['val_loss'].append(val_loss)
    history['val_acc'].append(val_acc)

    print(f"Epoch {epoch:02d}/{EPOCHS} [{epoch_duration:.1f}s] | "
          f"Train Loss: {train_loss:.4f}, Train Acc: {train_acc:.4f} | "
          f"Val Loss: {val_loss:.4f}, Val Acc: {val_acc:.4f}")

    # עדכון Scheduler (מבוסס על val_loss)
    if scheduler:
        scheduler.step(val_loss)

    # בדיקת שיפור ושמירת המודל הטוב ביותר
    if val_loss < best_val_loss:
        print(f"  Validation loss improved ({best_val_loss:.4f} --> {val_loss:.4f}). Saving model...")
        best_val_loss = val_loss
        # חשוב להשתמש ב-deepcopy כדי שהמשקולות לא ימשיכו להשתנות
        best_model_state = deepcopy(model.state_dict())
        all_val_labels = val_labels # שמירת תוויות וניבויים מה-epoch הטוב ביותר
        all_val_preds = val_preds
        epochs_no_improve = 0
        # שמירת המודל הטוב ביותר עד כה לקובץ
        torch.save({
            'epoch': epoch,
            'model_state_dict': best_model_state,
            'optimizer_state_dict': optimizer.state_dict(),
            'val_loss': best_val_loss,
            'val_acc': val_acc,
            'scaler': scaler, # שמירת ה-scaler ששימש לאימון!
            'hyperparameters': { # שמירת ההיפר-פרמטרים ששימשו
                 'lr': LEARNING_RATE, 'dropout': DROPOUT_P, 'wd': WEIGHT_DECAY,
                 'stat_dim': STAT_DIM, 'comb_dim': COMB_DIM,
                 'specaug_freq': SPECAUG_FREQ_MASK, 'specaug_time': SPECAUG_TIME_MASK,
                 'specaug_f_masks': SPECAUG_F_MASKS, 'specaug_t_masks': SPECAUG_T_MASKS
            }
        }, MODEL_FILENAME)

    else:
        epochs_no_improve += 1
        print(f"  Validation loss did not improve for {epochs_no_improve} epochs.")

    # עצירה מוקדמת
    if epochs_no_improve >= PATIENCE:
        print(f"\nEarly stopping triggered after {epoch} epochs.")
        break

total_training_time = time.time() - start_time
print(f"\n--- Training Finished ---")
print(f"Total Training Time: {total_training_time / 60:.2f} minutes")
print(f"Best Validation Loss: {best_val_loss:.4f}")
if best_model_state is None:
     print("WARNING: No best model state was saved. Training might have failed or stopped very early.")

# =====================================================================================
# =                    הצגת גרפים וסיכום                                             =
# =====================================================================================
if best_model_state is not None:
     print("\n--- Plotting Training Results ---")
     plot_training_results("FixedSplit", history, all_val_labels, all_val_preds)
else:
     print("\nSkipping plotting as no best model was found.")

# =====================================================================================
# =         אופציונלי: הערכה על נתונים חדשים (אם קיימים)                             =
# =====================================================================================
print("\n=== Loading and Evaluating on New External Dataset ===")
new_files = glob.glob(os.path.join(NEW_DATA_FOLDER, NEW_DATA_PATTERN))
if new_files and best_model_state is not None:
    new_data_map = {LABEL_STEPS: {'path': []}, LABEL_VEHICLE: {'path': []}, LABEL_NOISE: {'path': []}}
    # מיפוי גמיש יותר - נניח ששם הקובץ מציין את התווית
    for f in new_files:

        elif "nothing" in f or "noise" in f: new_data_map[LABEL_NOISE]['path'].append(f)
        else: print(f"Warning: Could not determine label for new data file: {f}")

    # הסר מחלקות ריקות מהמילון
    new_data_map = {k: v for k, v in new_data_map.items() if v['path']}

    if not new_data_map:

    else:

            # הצגת מטריצת בלבול ודו"ח סיווג לנתונים החדשים
            cm_new = confusion_matrix(new_true, new_pred, labels=range(NUM_CLASSES))

                                      labels=range(NUM_CLASSES), zero_division=0))

        else:
            print("No valid samples loaded from the new data folder.")
