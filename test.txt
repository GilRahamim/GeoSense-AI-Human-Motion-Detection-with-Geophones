import pandas as pd
import os
import math

print("--- Starting Manual Data Split Script (V2: Splitting 'nothing.csv') ---")

# ======================================================================
# =                       הגדרות ותצורה                              =
# ======================================================================

SOURCE_DATA_DIR = "."
OUTPUT_BASE_DIR = "fixed_split_data_v2" # <--- שם תיקייה חדש!
TRAIN_DIR_NAME = "train_data"
VALIDATION_DIR_NAME = "validation_data"

# יחס החלוקה לקבצים שיש לפצל
TRAIN_SPLIT_RATIO = 0.8 # 80% לאימון, 20% לוולידציה

# אסטרטגיית החלוקה החדשה:
SPLIT_CONFIG = {
    "train": [
        {"original_file": "man.csv", "output_file": "train_man.csv", "split": True},
        {"original_file": "car.csv", "output_file": "train_car.csv", "split": False}, # כל הקובץ
        {"original_file": "nothing.csv", "output_file": "train_nothing.csv", "split": True} # <-- נפצל את זה
    ],
    "validation": [
        {"original_file": "man.csv", "output_file": "validation_man.csv", "split": True},
        {"original_file": "car2.csv", "output_file": "validation_car2.csv", "split": False}, # כל הקובץ
        {"original_file": "nothing.csv", "output_file": "validation_nothing.csv", "split": True} # <-- נפצל את זה
        # לא נשתמש ב-nothing2.csv בכלל בניסוי הזה
    ]
}

# ======================================================================
# =                      יצירת תיקיות פלט                           =
# ======================================================================
train_output_dir = os.path.join(OUTPUT_BASE_DIR, TRAIN_DIR_NAME)
validation_output_dir = os.path.join(OUTPUT_BASE_DIR, VALIDATION_DIR_NAME)

try:
    os.makedirs(train_output_dir, exist_ok=True)
    os.makedirs(validation_output_dir, exist_ok=True)
    print(f"Output directories created/ensured:")
    print(f"  Train data: {train_output_dir}")
    print(f"  Validation data: {validation_output_dir}")
except OSError as e:
    print(f"Error creating directories: {e}")
    exit()
print("-" * 30)

# ======================================================================
# =                      תהליך הפיצול והשמירה                         =
# ======================================================================

processed_files_split_parts = {} # מילון לשמירת החלקים של הקבצים המפוצלים

# --- לולאה על כל קבצי המקור שדורשים פיצול ---
files_to_split = set()
for group in ["train", "validation"]:
    for item in SPLIT_CONFIG[group]:
        if item["split"]:
            files_to_split.add(item["original_file"])

print(f"Files identified for splitting: {list(files_to_split)}")

for original_filename in files_to_split:
    source_path = os.path.join(SOURCE_DATA_DIR, original_filename)
    print(f"  Pre-splitting '{original_filename}'...")

    if not os.path.exists(source_path):
        print(f"    ERROR: Source file not found: {source_path}. Cannot split.")
        continue

    try:
        df = pd.read_csv(source_path, header=None)
        if df.empty:
            print(f"    WARNING: Source file '{original_filename}' is empty. Skipping split.")
            processed_files_split_parts[original_filename] = {'train': df.iloc[0:0], 'validation': df.iloc[0:0]}
            continue

        total_rows = len(df)
        split_point = math.floor(total_rows * TRAIN_SPLIT_RATIO)

        if split_point == 0 or split_point >= total_rows:
            print(f"    WARNING: Invalid split point for {original_filename}. Check ratio or file length.")
            # במקרה כזה, נשים הכל ב-train (אם יש משהו) ונשאיר validation ריק
            df_train_part = df if split_point > 0 else df.iloc[0:0]
            df_val_part = df.iloc[0:0] # וולידציה ריקה
            print(f"    Assigning all {len(df_train_part)} rows to Train, 0 to Validation.")
        else:
            df_train_part = df.iloc[:split_point]
            df_val_part = df.iloc[split_point:]
            print(f"    Split: {len(df_train_part)} rows for Train, {len(df_val_part)} rows for Validation.")

        processed_files_split_parts[original_filename] = {
            'train': df_train_part,
            'validation': df_val_part
        }

    except Exception as e:
        print(f"    ERROR during pre-splitting of {original_filename}: {e}")


print("-" * 30)

# --- שמירת הקבצים המחולקים ---
for group, output_dir in [("train", train_output_dir), ("validation", validation_output_dir)]:
    print(f"Saving files for {group.upper()} set...")
    for item in SPLIT_CONFIG[group]:
        original_filename = item["original_file"]
        output_filename = item["output_file"]
        needs_split = item["split"]
        output_path = os.path.join(output_dir, output_filename)

        print(f"  Processing '{original_filename}' -> '{output_filename}'")

        try:
            # אם הקובץ פוצל מראש
            if needs_split:
                if original_filename in processed_files_split_parts:
                    df_to_save = processed_files_split_parts[original_filename][group]
                else:
                     print(f"    WARNING: File '{original_filename}' was marked for split but not found in pre-split parts. Skipping.")
                     continue
            # אם צריך לקחת את כל הקובץ המקורי (ולא פוצל)
            else:
                source_path = os.path.join(SOURCE_DATA_DIR, original_filename)
                if not os.path.exists(source_path):
                     print(f"    ERROR: Source file not found: {source_path}. Skipping.")
                     continue
                df_to_save = pd.read_csv(source_path, header=None)
                if df_to_save.empty:
                     print(f"    WARNING: Source file '{original_filename}' is empty. Creating empty output file.")
                     # ניצור קובץ ריק כדי שהטעינה לא תיכשל לגמרי
                     open(output_path, 'w').close()
                     continue

            # שמירת הקובץ
            if not df_to_save.empty:
                 df_to_save.to_csv(output_path, index=False, header=False)
                 print(f"    Saved {len(df_to_save)} rows to '{output_path}'")
            elif os.path.exists(output_path): # אם יצרנו קובץ ריק קודם
                 print(f"    Saved empty file to '{output_path}'")
            else: # אם ה-DataFrame ריק מהפיצול
                 open(output_path, 'w').close() # יצירת קובץ ריק
                 print(f"    Saved empty file to '{output_path}' (from split)")


        except Exception as e:
            print(f"    ERROR saving file {output_filename} for {group} set: {e}")

print("-" * 30)
print("--- Manual Data Split Script (V2) Finished ---")
